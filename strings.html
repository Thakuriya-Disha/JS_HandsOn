1) You are given a string S, and your task is to return an array B (having a size of 2), where B[0] contains the count of character A (uppercase) in string S and B[1] contains the count of character D (uppercase) in string S.

const countCharacters = (S) => 
{
  let countA=0; let countD=0;
  let N=S.length; let B=[];
  for(let i=0; i < N; i++)
  {
    if(S[i]==='A')
    countA++;
    if(S[i]==='D')
    countD++;
  }
  B[0]=countA;
  B[1]=countD;
  return B;
};

2) Tina is given a string S which contains the first letter of all the student names in her class. She got a curiosity to check how many people have their names starting from the same alphabet. So given a string S, she decided to write a code that finds out the count of characters that occur more than once in the string.

var Count_Occ = (s) => 
{
  let l=s.length;
  let newString = s.split("").sort().join("");
  let ans="";
  for(let i=0;i < l;i++)
  {
    let count=0;
    for(let j=0;j < l;j++)
    {
      if(newString[i]===newString[j] && i>j)
      {
        break;
      }
      if(newString[i]===newString[j])
      {
        count+=1;
      }
    }
    if(count>1)
    {
      ans+=newString[i]+count;
    }
  }
  return ans;
};

6) You are given a string S consisting of two letters A and D, where each character represent the winner of N games played between Aditya and Danish, where letter A represents the win of Aditya and letter D represents the win of Danish. var Game_Winner = (S) => 
{
  let n=S.length;
  let countA=0; let countD=0;
  for(let i=0; i < n; i++)
  {
    if(S[i]==='A')
    {
      countA++;
    }
    if(S[i]==='D')
    {
      countD++;
    }
  }
  if(countA===countD)
  {
    return 'Draw';
  }
  else
  {
    if(countA>countD)
    {
    return 'Aditya';
    }
  else
    {
    return 'Danish';
    }
  }
}

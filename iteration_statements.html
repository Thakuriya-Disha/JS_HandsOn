7) Write a program which takes a number from user and check whether number is prime number or not a prime number.

const Prime_Check = (N) => 
{
    if(N===1)
    {
        return 'NO';
    }
    else
    {
        for(let i=2; i<=Math.sqrt(N); ++i)
        {
            if(N%i===0)
            {
                return 'NO';
            }
        }
        return 'YES';
    }
};

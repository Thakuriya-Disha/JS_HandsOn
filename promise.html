1) Write one example explaining how you can write a callback function ? 

        function cubeFunc(num, operation){
            setTimeout(function(){
                let ans=1;
                ans = num**3;
                operation(num, ans);
            }, 2000)
        }
        
        cubeFunc(9, function(no, ans){
            console.log("The cube of ",no," is = ",ans);
        })
        

        //OUTPUT: The cube of 9 is = 729

    let count=0;
    setTimeout(function(){
        console.log(++count);
        setTimeout(function(){

                    console.log(++count);
                    setTimeout(function(){
